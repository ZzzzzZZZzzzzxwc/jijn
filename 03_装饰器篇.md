## 1. 装饰器基本概念

### 1.1 什么是装饰器

装饰器是一个函数，它接受一个函数作为输入，返回一个新函数。  
本质上是**高阶函数 + 闭包**的语法糖。

```python
# 装饰器的本质
def my_decorator(func):
    def wrapper():
        print("函数执行前")
        func()
        print("函数执行后")
    return wrapper

# 不用 @ 语法
def say_hello():
    print("Hello!")

say_hello = my_decorator(say_hello)  # 手动装饰
say_hello()

# 使用 @ 语法糖（等价写法）
@my_decorator
def say_hi():
    print("Hi!")

say_hi()
```

### 1.2 理解执行流程

```python
def decorator(func):
    print(f"装饰 {func.__name__}")  # 装饰时执行（导入时）
    def wrapper(*args, **kwargs):
        print("调用前")              # 调用时执行
        result = func(*args, **kwargs)
        print("调用后")              # 调用时执行
        return result
    return wrapper

@decorator
def greet(name):
    print(f"Hello, {name}!")

# 输出: 装饰 greet（导入模块时就执行了）

greet("小明")
# 输出:
# 调用前
# Hello, 小明!
# 调用后
```

## 2. 函数装饰器

### 2.1 处理带参数的被装饰函数

```python
def log_call(func):
    """记录函数调用"""
    def wrapper(*args, **kwargs):
        args_repr = [repr(a) for a in args]
        kwargs_repr = [f"{k}={v!r}" for k, v in kwargs.items()]
        signature = ", ".join(args_repr + kwargs_repr)
        print(f"调用 {func.__name__}({signature})")
        result = func(*args, **kwargs)
        print(f"返回 {result!r}")
        return result
    return wrapper

@log_call
def add(a, b):
    return a + b

add(3, 5)
# 调用 add(3, 5)
# 返回 8
```

### 2.2 计时装饰器

```python
import time

def timer(func):
    """计算函数执行时间"""
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        end = time.perf_counter()
        print(f"{func.__name__} 执行耗时: {end - start:.4f}秒")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(1)
    return "完成"

slow_function()  # slow_function 执行耗时: 1.0012秒
```

### 2.3 重试装饰器

```python
import time

def retry(max_attempts=3, delay=1):
    """失败自动重试"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts:
                        raise
                    print(f"第{attempt}次失败: {e}, {delay}秒后重试...")
                    time.sleep(delay)
        return wrapper
    return decorator

@retry(max_attempts=3, delay=0.5)
def unstable_api():
    import random
    if random.random() < 0.7:
        raise ConnectionError("连接失败")
    return "成功"
```

## 3. 带参数的装饰器

### 3.1 三层嵌套结构

```python
def repeat(times):
    """重复执行 n 次"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            results = []
            for _ in range(times):
                results.append(func(*args, **kwargs))
            return results
        return wrapper
    return decorator

@repeat(times=3)
def say(msg):
    print(msg)
    return msg

say("Hi!")
# Hi!
# Hi!
# Hi!
```

### 3.2 可选参数装饰器（带括号和不带括号都能用）

```python
def log(func=None, *, level="INFO"):
    """支持 @log 和 @log(level="DEBUG") 两种用法"""
    def decorator(fn):
        def wrapper(*args, **kwargs):
            print(f"[{level}] 调用 {fn.__name__}")
            return fn(*args, **kwargs)
        return wrapper
    
    if func is not None:
        # 不带参数调用: @log
        return decorator(func)
    # 带参数调用: @log(level="DEBUG")
    return decorator

@log
def func1():
    pass

@log(level="DEBUG")
def func2():
    pass

func1()  # [INFO] 调用 func1
func2()  # [DEBUG] 调用 func2
```

### 3.3 用类实现带参数装饰器

```python
class Retry:
    """用类实现的重试装饰器"""
    def __init__(self, max_attempts=3, exceptions=(Exception,)):
        self.max_attempts = max_attempts
        self.exceptions = exceptions
    
    def __call__(self, func):
        def wrapper(*args, **kwargs):
            for attempt in range(1, self.max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except self.exceptions as e:
                    if attempt == self.max_attempts:
                        raise
                    print(f"重试 {attempt}/{self.max_attempts}: {e}")
        return wrapper

@Retry(max_attempts=5, exceptions=(ValueError, ConnectionError))
def risky_operation():
    pass
```

## 4. 类装饰器

### 4.1 装饰类的函数

```python
def singleton(cls):
    """单例模式装饰器"""
    instances = {}
    def get_instance(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]
    return get_instance

@singleton
class Database:
    def __init__(self, url):
        self.url = url
        print(f"连接到 {url}")

db1 = Database("localhost:5432")  # 连接到 localhost:5432
db2 = Database("localhost:5432")  # 不再打印（返回同一实例）
print(db1 is db2)  # True
```

### 4.2 给类添加方法/属性

```python
def add_repr(cls):
    """自动添加 __repr__ 方法"""
    def __repr__(self):
        attrs = ", ".join(f"{k}={v!r}" for k, v in self.__dict__.items())
        return f"{cls.__name__}({attrs})"
    cls.__repr__ = __repr__
    return cls

@add_repr
class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

print(Point(3, 4))  # Point(x=3, y=4)
```

### 4.3 注册模式

```python
# 插件注册装饰器
registry = {}

def register(name):
    """注册类到全局字典"""
    def decorator(cls):
        registry[name] = cls
        return cls
    return decorator

@register("json")
class JsonParser:
    def parse(self, data):
        import json
        return json.loads(data)

@register("csv")
class CsvParser:
    def parse(self, data):
        return data.split(",")

# 工厂模式使用
def get_parser(format_name):
    return registry[format_name]()

parser = get_parser("json")
print(parser.parse('{"a": 1}'))  # {'a': 1}
```

### 4.4 用类作为装饰器（__call__ 方式）

```python
class CountCalls:
    """统计函数被调用次数"""
    def __init__(self, func):
        self.func = func
        self.count = 0
    
    def __call__(self, *args, **kwargs):
        self.count += 1
        print(f"{self.func.__name__} 已被调用 {self.count} 次")
        return self.func(*args, **kwargs)

@CountCalls
def say_hello():
    print("Hello!")

say_hello()  # say_hello 已被调用 1 次 \n Hello!
say_hello()  # say_hello 已被调用 2 次 \n Hello!
print(say_hello.count)  # 2
```

## 5. 装饰器叠加

```python
def bold(func):
    def wrapper(*args, **kwargs):
        return f"<b>{func(*args, **kwargs)}</b>"
    return wrapper

def italic(func):
    def wrapper(*args, **kwargs):
        return f"<i>{func(*args, **kwargs)}</i>"
    return wrapper

@bold
@italic
def greet(name):
    return f"Hello, {name}"

# 等价于: greet = bold(italic(greet))
# 执行顺序：从内到外装饰，从外到内执行
print(greet("World"))  # <b><i>Hello, World</i></b>
```

### 执行顺序详解

```python
def decorator_a(func):
    print("A 装饰")
    def wrapper(*args, **kwargs):
        print("A 前")
        result = func(*args, **kwargs)
        print("A 后")
        return result
    return wrapper

def decorator_b(func):
    print("B 装饰")
    def wrapper(*args, **kwargs):
        print("B 前")
        result = func(*args, **kwargs)
        print("B 后")
        return result
    return wrapper

@decorator_a
@decorator_b
def hello():
    print("Hello!")

# 装饰时输出（从内到外）:
# B 装饰
# A 装饰

hello()
# 调用时输出（从外到内）:
# A 前
# B 前
# Hello!
# B 后
# A 后
```

## 6. 保留函数元信息 functools.wraps

### 6.1 问题

```python
def my_decorator(func):
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@my_decorator
def greet(name):
    """打招呼函数"""
    return f"Hello, {name}"

print(greet.__name__)  # "wrapper" ← 丢失了原函数名！
print(greet.__doc__)   # None ← 丢失了文档！
```

### 6.2 解决方案

```python
from functools import wraps

def my_decorator(func):
    @wraps(func)  # ← 加上这一行
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@my_decorator
def greet(name):
    """打招呼函数"""
    return f"Hello, {name}"

print(greet.__name__)  # "greet" ✅
print(greet.__doc__)   # "打招呼函数" ✅
print(greet.__wrapped__)  # 可以访问原函数
```

### 6.3 标准装饰器模板

```python
from functools import wraps

def decorator(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        # 前置逻辑
        result = func(*args, **kwargs)
        # 后置逻辑
        return result
    return wrapper
```

### 7.5 @functools.cached_property (Python 3.8+)

```python
from functools import cached_property

class DataAnalysis:
    def __init__(self, data):
        self.data = data
    
    @cached_property
    def stats(self):
        """只计算一次，结果被缓存"""
        print("计算统计信息...")
        return {
            "mean": sum(self.data) / len(self.data),
            "max": max(self.data),
            "min": min(self.data)

analysis = DataAnalysis([1, 2, 3, 4, 5])
print(analysis.stats)  # 计算统计信息... {'mean': 3.0, ...}
print(analysis.stats)  # 直接返回缓存结果，不再计算
```

## 8. 实战装饰器模式

### 8.1 权限检查

```python
from functools import wraps

def require_auth(role="user"):
    """权限检查装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(user, *args, **kwargs):
            if not user.get("authenticated"):
                raise PermissionError("请先登录")
            if user.get("role") != role and role != "user":
                raise PermissionError(f"需要 {role} 权限")
            return func(user, *args, **kwargs)
        return wrapper
    return decorator

@require_auth(role="admin")
def delete_user(user, user_id):
    return f"已删除用户 {user_id}"

admin = {"authenticated": True, "role": "admin"}
guest = {"authenticated": True, "role": "user"}

print(delete_user(admin, 123))  # 已删除用户 123
# delete_user(guest, 123)       # ❌ PermissionError
```

### 8.2 缓存/记忆化

```python
from functools import wraps

def memoize(func):
    """通用缓存装饰器"""
    cache = {}
    
    @wraps(func)
    def wrapper(*args):
        if args not in cache:
            cache[args] = func(*args)
        return cache[args]
    
    wrapper.cache = cache
    wrapper.clear_cache = cache.clear
    return wrapper

@memoize
def expensive_calc(n):
    print(f"计算 {n}...")
    return n ** 2

print(expensive_calc(5))  # 计算 5... → 25
print(expensive_calc(5))  # 直接返回 25
```

### 8.3 输入验证

```python
from functools import wraps

def validate_types(**type_hints):
    """参数类型验证装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 检查位置参数
            import inspect
            sig = inspect.signature(func)
            params = list(sig.parameters.keys())
            
            for i, arg in enumerate(args):
                param_name = params[i]
                if param_name in type_hints:
                    expected = type_hints[param_name]
                    if not isinstance(arg, expected):
                        raise TypeError(
                            f"{param_name} 应为 {expected.__name__}, "
                            f"实际为 {type(arg).__name__}"
                        )
            return func(*args, **kwargs)
        return wrapper
    return decorator

@validate_types(name=str, age=int)
def create_user(name, age):
    return {"name": name, "age": age}

create_user("小明", 20)   # ✅
# create_user("小明", "20")  # ❌ TypeError: age 应为 int, 实际为 str
```

### 8.4 限流/节流

```python
import time
from functools import wraps

def rate_limit(max_calls, period):
    """限制函数在指定时间内的调用次数"""
    calls = []
    
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            now = time.time()
            # 清除过期记录
            while calls and calls[0] < now - period:
                calls.pop(0)
            
            if len(calls) >= max_calls:
                wait = period - (now - calls[0])
                raise RuntimeError(f"调用频率过高，请等待 {wait:.1f} 秒")
            
            calls.append(now)
            return func(*args, **kwargs)
        return wrapper
    return decorator

@rate_limit(max_calls=3, period=10)
def api_call():
    return "API 响应"
```

### 8.5 日志记录

```python
import logging
from functools import wraps

def log_execution(logger=None, level=logging.INFO):
    """函数执行日志装饰器"""
    def decorator(func):
        _logger = logger or logging.getLogger(func.__module__)
        
        @wraps(func)
        def wrapper(*args, **kwargs):
            _logger.log(level, f"开始执行 {func.__name__}")
            try:
                result = func(*args, **kwargs)
                _logger.log(level, f"{func.__name__} 执行成功")
                return result
            except Exception as e:
                _logger.error(f"{func.__name__} 执行失败: {e}")
                raise
        return wrapper
    return decorator
```

### 8.6 废弃警告

```python
import warnings
from functools import wraps

def deprecated(reason="", version=""):
    """标记函数为已废弃"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            msg = f"{func.__name__} 已废弃"
            if version:
                msg += f"(将在 {version} 移除)"
            if reason:
                msg += f": {reason}"
            warnings.warn(msg, DeprecationWarning, stacklevel=2)
            return func(*args, **kwargs)
        return wrapper
    return decorator
```

```python
def decorator_factory(arg1=None, arg2=None):
```

### 9.2 带状态的装饰器

```python
from functools import wraps

def count_calls(func):
    """统计调用次数"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        wrapper.calls += 1
        return func(*args, **kwargs)
    wrapper.calls = 0  # 在 wrapper 上存储状态
    return wrapper

@count_calls
def my_func():
    pass

my_func()
my_func()
print(my_func.calls)  # 2
```

```python
from functools import wraps

def debug_only(func):

    def wrapper(*args, **kwargs):
        if os.environ.get("DEBUG"):

    return wrapper
```

### 9.4 方法装饰器（处理 self）

```python
from functools import wraps

def validate_positive(method):
    """确保方法的第一个参数（除self外）为正数"""
    @wraps(method)
    def wrapper(self, value, *args, **kwargs):
        if value <= 0:
            raise ValueError(f"值必须为正数，收到: {value}")
        return method(self, value, *args, **kwargs)
    return wrapper

class BankAccount:
    def __init__(self, balance=0):
        self._balance = balance
    
    @validate_positive
    def deposit(self, amount):
        self._balance += amount
    
    @validate_positive
    def withdraw(self, amount):
        if amount > self._balance:
            raise ValueError("余额不足")
        self._balance -= amount
```

### 9.5 描述符实现装饰器

```python
class CachedProperty:
    """描述符实现的缓存属性"""
    def __init__(self, func):
        self.func = func
        self.attr_name = f"_cached_{func.__name__}"
    
    def __get__(self, obj, objtype=None):
        if obj is None:
            return self
        if not hasattr(obj, self.attr_name):
            setattr(obj, self.attr_name, self.func(obj))
        return getattr(obj, self.attr_name)

class Circle:
    def __init__(self, radius):
        self.radius = radius
    
    @CachedProperty
    def area(self):
        import math
        print("计算面积...")
        return math.pi * self.radius ** 2
```

## 总结：装饰器最佳实践

| 原则 | 说明 |
|------|------|
| 始终使用 `@wraps` | 保留被装饰函数的元信息 |
| 使用 `*args, **kwargs` | 让装饰器适用于任意参数的函数 |
| 保持单一职责 | 每个装饰器只做一件事 |
| 注意装饰顺序 | 从下往上装饰，从上往下执行 |
| 处理返回值 | 别忘了 `return func(...)` |
| 考虑性能 | 频繁调用的函数慎用复杂装饰器 |
| 写好文档 | 说明装饰器的作用和使用方式 |
