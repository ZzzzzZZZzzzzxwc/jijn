## 1. 类的基本定义

### 1.1 最简单的类

```python
class Dog:
    """狗类"""
    
    def __init__(self, name, age):
        """构造方法，初始化实例"""
        self.name = name  # 实例属性
        self.age = age
    
    def bark(self):
        """实例方法"""
        return f"{self.name}说: 汪汪汪!"
    
    def info(self):
        return f"{self.name}, {self.age}岁"

# 创建实例
dog1 = Dog("旺财", 3)
dog2 = Dog("小黑", 2)

print(dog1.bark())   # 旺财说: 汪汪汪!
print(dog2.info())   # 小黑, 2岁
```

### 1.2 `__init__` 不是构造函数

```python
class MyClass:
    def __new__(cls, *args, **kwargs):
        """真正的构造方法 - 创建实例"""
        print("__new__ 被调用")
        instance = super().__new__(cls)
        return instance
    
    def __init__(self, value):
        """初始化方法 - 初始化实例"""
        print("__init__ 被调用")
        self.value = value

obj = MyClass(42)
# 输出:
# __new__ 被调用
# __init__ 被调用
```

## 2. 实例属性与类属性

### 2.1 区别

```python
class Student:
    # 类属性 - 所有实例共享
    school = "Python大学"
    count = 0
    
    def __init__(self, name):
        # 实例属性 - 每个实例独有
        self.name = name
        Student.count += 1

s1 = Student("小明")
s2 = Student("小红")

print(Student.count)   # 2 (类属性)
print(s1.school)       # Python大学 (通过实例访问类属性)
print(s1.name)         # 小明 (实例属性)
```

### 2.2 属性查找顺序

```python
class Example:
    x = "类属性"
    
    def __init__(self):
        self.x = "实例属性"

obj = Example()
print(obj.x)      # "实例属性" (实例属性优先)
print(Example.x)  # "类属性"
del obj.x
print(obj.x)      # "类属性" (删除实例属性后回退到类属性)
```

### 2.3 `__slots__` 限制属性

```python
class Point:
    __slots__ = ('x', 'y')  # 只允许这些属性
    
    def __init__(self, x, y):
        self.x = x
        self.y = y

p = Point(1, 2)
# p.z = 3  # ❌ AttributeError: 'Point' object has no attribute 'z'

# 好处: 节省内存, 访问更快
```

## 3. 方法类型

### 3.1 实例方法

```python
class MyClass:
    def instance_method(self):
        """第一个参数是 self (实例本身)"""
        return f"实例方法, self={self}"
```

### 3.2 类方法 @classmethod

```python
class Date:
    def __init__(self, year, month, day):
        self.year = year
        self.month = month
        self.day = day
    
    @classmethod
    def from_string(cls, date_string):
        """类方法：替代构造器"""
        year, month, day = map(int, date_string.split('-'))
        return cls(year, month, day)  # cls 是类本身
    
    @classmethod
    def today(cls):
        """类方法：获取今天的日期"""
        import datetime
        t = datetime.date.today()
        return cls(t.year, t.month, t.day)

# 使用类方法创建实例
d1 = Date(2024, 1, 15)
d2 = Date.from_string("2024-06-20")
d3 = Date.today()
```

### 3.3 静态方法 @staticmethod

```python
class MathUtils:
    @staticmethod
    def is_even(n):
        """静态方法：不需要访问实例或类"""
        return n % 2 == 0
    
    @staticmethod
    def gcd(a, b):
        """最大公约数"""
        while b:
            a, b = b, a % b
        return a

# 可以通过类或实例调用
print(MathUtils.is_even(4))   # True
print(MathUtils.gcd(12, 8))   # 4
```

### 3.4 三种方法对比

```python
class Demo:
    class_var = "类变量"
    
    def __init__(self):
        self.instance_var = "实例变量"
    
    def instance_method(self):
        # 可以访问实例和类
        return self.instance_var + " | " + self.class_var
    
    @classmethod
    def class_method(cls):
        # 只能访问类，不能访问实例
        return cls.class_var
    
    @staticmethod
    def static_method():
        # 既不能访问实例，也不能访问类
        return "独立的工具函数"
```

## 4. 继承

### 4.1 单继承

```python
class Animal:
    def __init__(self, name):
        self.name = name
    
    def speak(self):
        raise NotImplementedError("子类必须实现 speak 方法")

class Dog(Animal):
    def speak(self):
        return f"{self.name}: 汪汪!"

class Cat(Animal):
    def speak(self):
        return f"{self.name}: 喵喵~"

dog = Dog("旺财")
cat = Cat("咪咪")
print(dog.speak())  # 旺财: 汪汪!
print(cat.speak())  # 咪咪: 喵喵~
```

### 4.2 super() 调用父类方法

```python
class Base:
    def __init__(self, x):
        self.x = x
        print(f"Base.__init__({x})")

class Child(Base):
    def __init__(self, x, y):
        super().__init__(x)  # 调用父类的 __init__
        self.y = y
        print(f"Child.__init__({x}, {y})")

c = Child(1, 2)
# Base.__init__(1)
# Child.__init__(1, 2)
```

### 4.3 多继承

```python
class A:
    def method(self):
        return "A"

class B(A):
    def method(self):
        return "B"

class C(A):
    def method(self):
        return "C"

class D(B, C):
    pass

d = D()
print(d.method())  # "B" (根据 MRO 顺序)

# 查看方法解析顺序 (MRO)
print(D.__mro__)
# (<class 'D'>, <class 'B'>, <class 'C'>, <class 'A'>, <class 'object'>)
```

### 4.4 isinstance 和 issubclass

```python
print(isinstance(dog, Dog))     # True
print(isinstance(dog, Animal))  # True (Dog 继承自 Animal)
print(issubclass(Dog, Animal))  # True
print(issubclass(Dog, Cat))     # False
```

## 5. 多态

```python
class Shape:
    def area(self):
        raise NotImplementedError

class Rectangle(Shape):
    def __init__(self, w, h):
        self.w, self.h = w, h
    def area(self):
        return self.w * self.h

class Circle(Shape):
    def __init__(self, r):
        self.r = r
    def area(self):
        import math
        return math.pi * self.r ** 2

class Triangle(Shape):
    def __init__(self, base, height):
        self.base, self.height = base, height
    def area(self):
        return 0.5 * self.base * self.height

# 多态：统一接口，不同实现
shapes = [Rectangle(3, 4), Circle(5), Triangle(6, 8)]
for shape in shapes:
    print(f"{shape.__class__.__name__}: 面积 = {shape.area():.2f}")

# 鸭子类型 - Python 不需要严格继承
class CustomShape:
    def area(self):
        return 42

shapes.append(CustomShape())  # 只要有 area() 方法就行
```

## 6. 封装与访问控制

```python
class BankAccount:
    def __init__(self, owner, balance=0):
        self.owner = owner          # 公开属性
        self._balance = balance     # 受保护属性 (约定，不强制)
        self.__pin = "1234"         # 私有属性 (名称改写)
    
    def deposit(self, amount):
        if amount > 0:
            self._balance += amount
    
    def withdraw(self, amount):
        if 0 < amount <= self._balance:
            self._balance -= amount
            return amount
        return 0
    
    def get_balance(self):
        return self._balance

acc = BankAccount("小明", 1000)
print(acc.owner)         # ✅ 公开
print(acc._balance)      # ⚠️ 可以访问但不建议
# print(acc.__pin)       # ❌ AttributeError
print(acc._BankAccount__pin)  # ⚠️ 名称改写后仍可访问 "1234"
```

## 7. 属性装饰器 property

```python
class Temperature:
    def __init__(self, celsius=0):
        self._celsius = celsius
    
    @property
    def celsius(self):
        """getter"""
        return self._celsius
    
    @celsius.setter
    def celsius(self, value):
        """setter - 带验证"""
        if value < -273.15:
            raise ValueError("温度不能低于绝对零度!")
        self._celsius = value
    
    @celsius.deleter
    def celsius(self):
        """deleter"""
        print("删除温度值")
        del self._celsius
    
    @property
    def fahrenheit(self):
        """只读属性（没有 setter）"""
        return self._celsius * 9/5 + 32

t = Temperature(25)
print(t.celsius)      # 25 (调用 getter)
print(t.fahrenheit)   # 77.0
t.celsius = 30        # 调用 setter
# t.celsius = -300    # ❌ ValueError
```

## 8. 魔术方法 (Dunder Methods)

### 8.1 字符串表示

```python
class Point:
    def __init__(self, x, y):
        self.x, self.y = x, y
    
    def __repr__(self):
        """开发者看的表示（可用 eval 重建对象）"""
        return f"Point({self.x}, {self.y})"
    
    def __str__(self):
        """用户看的表示"""
        return f"({self.x}, {self.y})"

p = Point(3, 4)
print(repr(p))  # Point(3, 4)
print(str(p))   # (3, 4)
print(p)        # (3, 4) ← print 调用 __str__
```

### 8.2 运算符重载

```python
class Vector:
    def __init__(self, x, y):
        self.x, self.y = x, y
    
    def __add__(self, other):       # +
        return Vector(self.x + other.x, self.y + other.y)
    
    def __sub__(self, other):       # -
        return Vector(self.x - other.x, self.y - other.y)
    
    def __mul__(self, scalar):      # *
        return Vector(self.x * scalar, self.y * scalar)
    
    def __rmul__(self, scalar):     # 反向 * (scalar * vector)
        return self.__mul__(scalar)
    
    def __eq__(self, other):        # ==
        return self.x == other.x and self.y == other.y
    
    def __abs__(self):              # abs()
        return (self.x**2 + self.y**2) ** 0.5
    
    def __bool__(self):             # bool()
        return self.x != 0 or self.y != 0
    
    def __repr__(self):
        return f"Vector({self.x}, {self.y})"

v1 = Vector(3, 4)
v2 = Vector(1, 2)
print(v1 + v2)     # Vector(4, 6)
print(v1 * 3)      # Vector(9, 12)
print(3 * v1)      # Vector(9, 12)
print(abs(v1))     # 5.0
```

### 8.3 比较运算符

```python
from functools import total_ordering

@total_ordering  # 只需定义 __eq__ 和 __lt__，自动生成其他比较方法
class Student:
    def __init__(self, name, score):
        self.name = name
        self.score = score
    
    def __eq__(self, other):
        return self.score == other.score
    
    def __lt__(self, other):
        return self.score < other.score

s1 = Student("小明", 85)
s2 = Student("小红", 92)
print(s1 < s2)   # True
print(s1 >= s2)  # False (自动生成)
```

### 8.4 容器协议

```python
class Deck:
    """扑克牌组"""
    ranks = list(range(2, 11)) + ['J', 'Q', 'K', 'A']
    suits = ['♠', '♥', '♦', '♣']
    
    def __init__(self):
        self.cards = [f"{r}{s}" for s in self.suits for r in self.ranks]
    
    def __len__(self):              # len()
        return len(self.cards)
    
    def __getitem__(self, index):   # deck[i] 和切片
        return self.cards[index]
    
    def __contains__(self, card):   # in 运算符
        return card in self.cards
    
    def __iter__(self):             # for 循环
        return iter(self.cards)

deck = Deck()
print(len(deck))         # 52
print(deck[0])           # 2♠
print(deck[-1])          # A♣
print("A♠" in deck)      # True
print(deck[0:3])         # ['2♠', '3♠', '4♠']
```

### 8.5 上下文管理器

```python
class FileManager:
    def __init__(self, filename, mode):
        self.filename = filename
        self.mode = mode
        self.file = None
    
    def __enter__(self):
        """进入 with 块时调用"""
        self.file = open(self.filename, self.mode)
        return self.file
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        """退出 with 块时调用"""
        if self.file:
            self.file.close()
        return False  # 不抑制异常

# 使用
with FileManager("test.txt", "w") as f:
    f.write("Hello!")
```

### 8.7 常用魔术方法速查表

| 方法 | 触发方式 | 说明 |
|------|---------|------|
| `__init__` | `Class()` | 初始化 |
| `__new__` | `Class()` | 创建实例 |
| `__del__` | `del obj` | 析构 |
| `__repr__` | `repr(obj)` | 开发者表示 |
| `__str__` | `str(obj)` | 用户表示 |
| `__len__` | `len(obj)` | 长度 |
| `__getitem__` | `obj[key]` | 取值 |
| `__setitem__` | `obj[key]=v` | 赋值 |
| `__delitem__` | `del obj[key]` | 删除 |
| `__contains__` | `x in obj` | 包含 |
| `__iter__` | `for x in obj` | 迭代 |
| `__next__` | `next(obj)` | 下一个 |
| `__call__` | `obj()` | 调用 |
| `__hash__` | `hash(obj)` | 哈希 |
| `__eq__` | `obj == other` | 相等 |
| `__lt__` | `obj < other` | 小于 |
| `__add__` | `obj + other` | 加 |
| `__enter__/__exit__` | `with obj` | 上下文 |
| `__getattr__` | `obj.x` (属性不存在时) | 属性访问 |
| `__setattr__` | `obj.x = v` | 属性设置 |

## 9. 抽象类与接口

```python
from abc import ABC, abstractmethod

class Shape(ABC):
    """抽象基类 - 不能直接实例化"""
    
    @abstractmethod
    def area(self):
        """子类必须实现"""
        pass
    
    @abstractmethod
    def perimeter(self):
        pass
    
    def description(self):
        """普通方法 - 子类可以继承"""
        return f"{self.__class__.__name__}: 面积={self.area():.2f}"

# shape = Shape()  # ❌ TypeError: Can't instantiate abstract class

class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius
    
    def area(self):
        import math
        return math.pi * self.radius ** 2
    
    def perimeter(self):
        import math
        return 2 * math.pi * self.radius

c = Circle(5)
print(c.description())  # Circle: 面积=78.54
```

## 10. 元类 (Metaclass)

### 10.1 基本概念

```python
# 类是元类的实例，默认元类是 type
class MyClass:
    pass

print(type(MyClass))       # <class 'type'>
print(type(MyClass()))     # <class 'MyClass'>

# 用 type 动态创建类
Dog = type('Dog', (object,), {'speak': lambda self: '汪!'})
d = Dog()
print(d.speak())  # 汪!
```

### 10.2 自定义元类

```python
class SingletonMeta(type):
    """单例元类"""
    _instances = {}
    
    def __call__(cls, *args, **kwargs):
        if cls not in cls._instances:
            cls._instances[cls] = super().__call__(*args, **kwargs)
        return cls._instances[cls]

class Database(metaclass=SingletonMeta):
    def __init__(self):
        self.connection = "Connected"

db1 = Database()
db2 = Database()
print(db1 is db2)  # True (同一个实例)
```

## 11. 数据类 dataclass

```python
from dataclasses import dataclass, field

@dataclass
class Point:
    x: float
    y: float

@dataclass(order=True)  # 自动生成比较方法
class Student:
    name: str
    score: float
    courses: list = field(default_factory=list)  # 可变默认值
    
    # 排序按 sort_index 字段
    sort_index: float = field(init=False, repr=False)
    
    def __post_init__(self):
        self.sort_index = self.score

# 自动生成 __init__, __repr__, __eq__
p = Point(3, 4)
print(p)  # Point(x=3, y=4)

s1 = Student("小明", 85)
s2 = Student("小红", 92)
print(s1 < s2)  # True (按 score 比较)

# 不可变数据类
@dataclass(frozen=True)
class Color:
    r: int
    g: int
    b: int

red = Color(255, 0, 0)
# red.r = 128  # ❌ FrozenInstanceError
```

## 12. Mixin 与多继承

```python
# Mixin: 提供可复用的功能片段
class SerializeMixin:
    """序列化能力"""
    def to_dict(self):
        return self.__dict__.copy()
    
    def to_json(self):
        import json
        return json.dumps(self.to_dict(), ensure_ascii=False)

class LogMixin:
    """日志能力"""
    def log(self, message):
        print(f"[{self.__class__.__name__}] {message}")

class ValidateMixin:
    """验证能力"""
    def validate(self):
        for key, value in self.__dict__.items():
            if value is None:
                raise ValueError(f"{key} 不能为 None")

# 组合使用
class User(SerializeMixin, LogMixin, ValidateMixin):
    def __init__(self, name, email):
        self.name = name
        self.email = email

user = User("小明", "xm@example.com")
user.log("创建用户")           # [User] 创建用户
print(user.to_json())          # {"name": "小明", "email": "xm@example.com"}
user.validate()                # 通过验证
```
